Recursive constructor for the win ! 

  
class FancyHash():
    def __init__(self, a_dictonary):
        for key, value in a_dictonary.items():
            if instanceof(value, dict):
                self.__dict__[key] = FancyHash(value)
            else:
                self.__dict__[key] = value

db.py 

  
from django.contrib.auth.models import User
from api.models import School, DaClass, ClassSession
import datetime
from changeless.decorators import fancy_list, fancy_gen

@fancy_list(depth=0)
def all_schools():
    return School.objects.all()

@fancy_gen
def all_classes():
    return DaClass.objects.all()

@fancy_list
def todays_classes():
    today_start = datetime.date.today()
    today_end = datetime.date.today() + datetime.timedelta(hours=24)
    return ClassSession.objects.filter(start_time__gte=today_start, start_time__lte=today_end)

@fancy_list
def check_in(user_id, session_id):
    session = ClassSession.objects.get(session_id)
    user = User.objects.get(user_id)
    session.check_ins.add(user)
    session.save()
    return session

@fancy_gen
def class_sessions(class_id):
    return ClassSession.objects.filter(da_class__id==class_id)

Global State

Set data-state="something" on a slide and "something" will be added as a class to the document element when the slide is open. This lets you apply broader style changes, like switching the background.

Custom Events

Additionally custom events can be triggered on a per slide basis by binding to the data-state name. 


Reveal.addEventListener( 'customevent', function() {
    console.log( '"customevent" has fired' );
} );

Slide Backgrounds

Set data-background="#007777" on a slide to change the full page background to the given color. All CSS color formats are supported.

Image Backgrounds

Background Transitions

Pass reveal.js the backgroundTransition: 'slide' config argument to make backgrounds slide rather than fade.

Background Transition Override

You can override background transitions per slide by using data-background-transition="slide".
