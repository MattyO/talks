Recursive constructor for the win ! 

  
class FancyHash():
    def __init__(self, a_dictonary):
        for key, value in a_dictonary.items():
            if instanceof(value, dict):
                self.__dict__[key] = FancyHash(value)
            else:
                self.__dict__[key] = value

from django.contrib.auth.models import User
from api.models import School, DaClass, ClassSession
import datetime
from changeless.decorators import fancy_list, fancy_gen

@fancy_list(depth=0)
def all_schools():
    return School.objects.all()

@fancy_gen
def all_classes():
    return DaClass.objects.all()

@fancy_list
def todays_classes():
    today_start = datetime.date.today()
    today_end = datetime.date.today() + datetime.timedelta(hours=24)
    return ClassSession.objects.filter(start_time__gte=today_start, start_time__lte=today_end)

@fancy_list
def check_in(user_id, session_id):
    session = ClassSession.objects.get(session_id)
    user = User.objects.get(user_id)
    session.check_ins.add(user)
    session.save()

@fancy_gen
def class_sessions(class_id):
    return ClassSession.objects.filter(da_class__id==class_id)
